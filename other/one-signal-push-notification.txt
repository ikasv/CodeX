 public function onesignalNotification3($notify_users, $sendData){
        
        $setting        =   General_setting::first();
        $content        =   array( "en" => $sendData['body'] );
        
        $curl           =   curl_init();
        
          $fields = array(
            'app_id'                =>  "$setting->os_app_id",
            "headings"              =>  array( "en" => $sendData['title'] ),
            'include_player_ids'    =>  array_unique($notify_users),
            'large_icon'            => "https://banshiclasses.com/admin/public/assets/images/admin/1617688296.jpeg",
            'content_available'     => true,
            'contents'              => $content,
        );
        
        // "included_segments"     :   ["Subscribed Users"]}',
        
        curl_setopt_array($curl, array(
                                      CURLOPT_URL => 'https://onesignal.com/api/v1/notifications',
                                      CURLOPT_RETURNTRANSFER => true,
                                      CURLOPT_ENCODING => '',
                                      CURLOPT_MAXREDIRS => 10,
                                      CURLOPT_TIMEOUT => 0,
                                      CURLOPT_FOLLOWLOCATION => true,
                                      CURLOPT_HTTP_VERSION => CURL_HTTP_VERSION_1_1,
                                      CURLOPT_CUSTOMREQUEST => 'POST',
                                      CURLOPT_POSTFIELDS =>     json_encode($fields),
                                      CURLOPT_HTTPHEADER => array(
                                        'Content-Type: application/json; charset=utf-8',
                                        'Authorization: Basic ZDJjZWZmYWMtNGZjYy00YWNkLWFjZWYtZTY1YzgwOTU3MWZj'
                                      ),
        ));

        $response       =   curl_exec($curl);

        curl_close($curl);
        return $response;
    }
