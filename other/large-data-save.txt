public function hotels_with_details(){
        ini_set('memory_limit', '9999M');
        set_time_limit(9000);

        $result                 =   HotelFetchAPI()->hotelList();

        // $hotels_with_details    =   array();

        // $path                   =   public_path('json_files/hotel/hotels-with-details.json');
        // $file = fopen($path, "c");
        // fseek($file, -3, SEEK_END);

        $chunk_hotels       = array_chunk($result['hotels'], 200);

        
        foreach($chunk_hotels as $hotels):
          foreach($hotels as $hotel):
                  
              $curl = curl_init();

              curl_setopt_array($curl, array(
                  CURLOPT_URL            => $this->base_url . 'GetListing?id=' . $hotel->id,
                  CURLOPT_RETURNTRANSFER => true,
                  CURLOPT_ENCODING       => '',
                  CURLOPT_MAXREDIRS      => 10,
                  CURLOPT_TIMEOUT        => 1000,
                  CURLOPT_FOLLOWLOCATION => true,
                  CURLOPT_HTTP_VERSION   => CURL_HTTP_VERSION_1_1,
                  CURLOPT_CUSTOMREQUEST  => 'GET',
                  CURLOPT_HTTPHEADER     => array
                      (
                          $this->auth,
                          $this->cache_control,
                          $this->xml_type
                      ),
                  ));

                  $response  =  curl_exec($curl);

                  curl_close($curl);
                  // $hotels_with_details[]      =   array('hotel_id' =>  $hotel->id, 'details' => $response);
                  Hotel::insert(array('hotel_id' =>  $hotel->id, 'details' => $response));
                  
                  // fwrite($file, json_encode($hotels_with_details));
              endforeach;
            endforeach;
        // fclose($file);

        // dd($hotels_with_details);
       
    }
