<?php

namespace App\View\Components;

use Illuminate\View\Component;

class BaggageAndFareRulesInfoModal extends Component
{
    public $baggageId;
    public $fareRuleId;
    public $fareRuleInfo;
    public $baggageInfo;
    
    public function __construct($baggageId, $fareRuleId, $fareRuleInfo, $baggageInfo)
    {
        $this->baggageId        =   $baggageId;
        $this->fareRuleId       =   $fareRuleId;
        $this->fareRuleInfo       =   $fareRuleInfo;
        $this->baggageInfo       =   $baggageInfo;
    }
    
    public function baggageDetails(){
        return $this->baggageInfo->Response->Results->Segments[0];
    }
    
    public function fareRuleDetails(){
        return $this->fareRuleInfo->Response->FareRules;
    }
    
    public function render()
    {
        return view('components.baggage_and_fare_rules_info_modal');
    }
}
?>
