public function getZohoAccessToken()
{
$url="https://accounts.zoho.com/oauth/v2/token";

$data_tok = array('refresh_token' => REFRESH_TOKEN, 'client_id' => CLIENT_ID,'client_secret' => CLIENT_SECRET,'grant_type' => 'refresh_token');

$options = array(

'http' => array(

'header'  => "Content-type: application/x-www-form-urlencoded\r\n",

'method'  => 'POST',

'content' => http_build_query($data_tok),

),
'ssl' => array(
    "verify_peer" => false,
    "verify_peer_name" => false,
)

);



$context  = stream_context_create($options);

$result = file_get_contents($url, false, $context);

$data1=json_decode($result,true);

$token=$data1["access_token"];

return $token;
}
