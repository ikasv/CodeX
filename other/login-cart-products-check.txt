                        $session_cart_items_arr[]     =   $session_cart_item->id;
                    endforeach;

                # Update Cart
                $user_cart_itmes   =     getCartItems();
                
                foreach($user_cart_itmes->cart_items as $user_cart_itme):
                    if(in_array($user_cart_itme->id, $session_cart_items_arr)):
                        remove_cart_item($user_cart_itme->id, false);
                    endif;
                endforeach;
