if (!function_exists('sendMail')) {
    function sendMail($to = 'sudhir@honeycombindia.net', $mailsubject = 'Mail Subject', $content = '<p>html body</p>', $mailfrom = '' ){
            
            Mail::send([],[], function ($mail) use ($mailsubject, $content, $to)
            {
                $mail->to($to)
                    ->subject($mailsubject)
                    ->setBody($content, 'text/html');
                    $mail->from('no-reply@honeycombindia.net', 'Your little list');
            });
            
            return true;
    }
}
