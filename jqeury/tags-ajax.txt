<script src="https://cdnjs.cloudflare.com/ajax/libs/jquery-validate/1.19.5/jquery.validate.min.js" integrity="sha512-rstIgDs0xPgmG6RX1Aba4KV5cWJbAMcvRCVmglpam9SoHZiUCyQVDdH2LPlxoHtrv17XWblE/V/PP+Tr04hbtA==" crossorigin="anonymous" referrerpolicy="no-referrer"></script>
<script>
        var table = $('#tags-table').DataTable();
    $(document).ready(function(){
        
        // Close Modal
        $('.btn-close').click(function(){
            $('#tag-form')[0].reset();
            $('#type').val('insert');
            $('#tag-id').val('');   
            $('.ajax-msg').html('');
        });
        // End Close Modal
        
        $("#tag-form").validate({
        rules: {
            name: {
                required: true,
            },
        },
        messages: {
            name: {
              required: "Please enter a tag name",
        }
    },
    submitHandler: function(form) {
        var expform     =   document.getElementById('tag-form');
        var fd          =   new FormData(expform);
        $.ajax({
            type: "POST",
            url: "<?= home_asset(); ?>/ajax/tag.php",
            data: fd,
            dataType: 'json',
            contentType: false,
            processData: false,
            beforeSend: function() {
                $('.submit-btn').prop('disabled', true);
            },
            success: function(data) {
                $('#tag-form')[0].reset();
                $('.submit-btn').prop('disabled', false);
                
                if(data.status == true){
                    $('.ajax-msg').html('<div class="alert alert-success">'+data.message+'<div>');
                }
                
                if(data.status == false){
                    $('.ajax-msg').html('<div class="alert alert-danger">'+data.message+'<div>');
                }
                
              reload_page();
                
            },
            error: function() {
                alert('Some Error Occured.');
            }
        });
    }
});
        
        
    });
    
    // Edit Tag
         function editTag(id){
             
                $('#type').val('update');
             
                $.ajax({
                    type: "POST",
                    url: "<?= home_asset(); ?>/ajax/tag.php",
                    data: {type:'single', id:id},
                    dataType: 'json',
                    success: function(data) {
                        if(data.status == true){
                            $('#tag-id').val(id);   
                            $('#tag-name').val(data.data.name); 
                            $('#tagModal').modal('show');
                        }
                        
                        if(data.status == false){
                            alert("some error occured");
                        }
                    },
                    error: function() {
                        alert('Some Error Occured.');
                    }
            });   
        }
        // End Edit Tag
        
        // Delete Tag
        function deleteTag(id){
            
            if(confirm("Are you sure!")){
                
                $.ajax({
                    type: "POST",
                    url: "<?= home_asset(); ?>/ajax/tag.php",
                    data: {type:'delete', id:id},
                    dataType: 'json',
                    success: function(data) {
                        if(data.status == true){
                            alert(data.message);
                           reload_page();
                        }
                        
                        if(data.status == false){
                            alert("some error occured");
                        }
                    },
                    error: function() {
                        alert('Some Error Occured.');
                    }
                });  
            
            }
        }
        // End Delete Tag
        
        function reload_page(){
                            setTimeout(function(){
                      location.reload();
                },500);
        }
</script>
