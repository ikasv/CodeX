    /*
    *   @param tmp name of file 
    *   return boolean 
    */
    function isMalformedFiles($tmp){
        $contents = htmlspecialchars(file_get_contents($tmp));
        $pattern = "/<?php|echo|script|document.|script&gt;/im";

        if (preg_match_all($pattern, $contents, $matches)):
            return true;
        else:
            return false;
        endif;
    }
