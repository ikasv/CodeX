 	if (!function_exists('custom_pagination')) {
    function custom_pagination($items, $perPage = 1, $currentPage = 1){
        
        $offSet = ($currentPage * $perPage) - $perPage;
        
        if(count($items) > $perPage):
            $itemsForCurrentPage = array_slice($items, $offSet, $perPage, true);
        endif;
        
        $items_with_pagination = new LengthAwarePaginator(array_values($itemsForCurrentPage ?? $items), count($items), $perPage, $currentPage);
        return $items_with_pagination;
    }
 }
