    function isAnyPostValueMalformed($post_data){
            $pattern = "/<?php|<script>|&lt;script&gt;/im";

            foreach ($post_data as $key => $value) {
                if (preg_match_all($pattern, $value, $matches)){
                    return true;
                }
             }
        return false;
    }
 
    if(isset($_POST['submit'])):
        if(isAnyPostValueMalformed($_POST)):
           die('Malformed found in value');
        endif;
    endif;
