    # Email Functionality
    private function sendEmail($message, $from, $to, $subject, $siteName = "")
    {
        $this->email->set_mailtype("html");
        $this->email->set_newline("\r\n");
        $this->email->from($from ,$siteName);
        $this->email->to($to);
        $this->email->subject($subject);
        $this->email->message($message);
        $this->email->send();
    }
