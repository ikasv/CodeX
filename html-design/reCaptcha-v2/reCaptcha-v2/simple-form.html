Step 4 : Verified Google reCaptcha 

    public function submit_form"()
    {
        $this->load->library('session');

        if($this->input->post())
        {
            # reCaptcha
            $recaptchaResponse  = trim($this->input->post('g-recaptcha-response'));
            $userIp=$this->input->ip_address();
            $secret             = $this->config->item('Secret Key Here');
            $url  = "https://www.google.com/recaptcha/api/siteverify?secret=".$secret."&response=".$recaptchaResponse."&remoteip=".$userIp;
            $ch                 = curl_init(); 
            curl_setopt($ch, CURLOPT_URL, $url); 
            curl_setopt($ch, CURLOPT_RETURNTRANSFER, 1); 
            $output             = curl_exec($ch); 
            curl_close($ch);      
            $status             = json_decode($output, true);
            # End reCaptcha

            if ($status['success']) 
            {
                    $data = array(
                    'name'        => $this->input->post('name'),
                    'email'       => $this->input->post('email'),
                    'mobile'      => $this->input->post('mobile'),
                    'address'     => $this->input->post('address')
                    );

                $result =  $this->Action_model->insert_data($data,'tbl_contact');
                    if($result)
                    {
                        $this->session->set_flashdata('session_msg','<div class="alert alert-success" role="alert">Thanks for your feedback.</div>');
                        redirect('contact');
                    }
                    else
                    {
                        $this->session->set_flashdata('session_msg', '<div class="alert alert-danger" role="alert">Failed to submit. Try again.</div>');
                        redirect('contact');    
                    }
            }
            else
            { 
                $this->session->set_flashdata('session_msg', '<div class="alert alert-danger" role="alert">Please Check Google Captcha 
                to submit form. Try again.</div>');
                redirect('contact');  
            }
        }   
    }
