    /* Video */
    var sr_no = 1;
    function addVideo(no){
        var length = $(".add-video-row").last().attr('data-row');
        let i = ++length;
        var cloned = $('.add-video-row').last().clone();
        $('.add-video-wrapper').append(cloned);
        $(".add-video-row").last().attr('data-row', i);
        $(".add-video-row").last().addClass('video_row_'+i);
        $(".add-video-wrapper .add-video-row .remove-btn").last().attr('data-remove', i);
        $(".add-video-row").last().find('.sr_no').val(i);
    }
        
           $(document).on("click", ".remove-video-btn", function(){
            $rm_id = $(this).attr('data-remove');
            // alert("video_row_"+$rm_id);
            $(".add-video-wrapper .video_row_"+$rm_id).remove();
       });
    /* Video End*/
