# AJAX GET STATE
function country_get(id = '') {
    $.ajax({
        type: "POST",
        url: "<?php echo base_url('').'action/get_state'; ?>",
        data: {
            id: id
        },
        dataType: 'json',
        success: function(data) {
            if (data.status == 'true') {
                var state_list = data.state_list;
                var row = "<option value=''>Select State</option>";
                for (var i = 0; i < state_list.length; i++) {
                    row += "<option class='text-capitalize' value='" + state_list[i].state_id +
                        "'>" +
                        state_list[i].name + "</option>";
                }
                $("#state").html(row);
            } else {
                $("#state").html("<option value=''>Select State</option>");
            }
        },
        error: function() {
            alert('Some Error Occured.');
        }

    });
}
# End GET STATE

    public function get_state()
    {
        $arr = array();
        $state_list = $this->db->get_where('states', array("country_id" => $this->input->post('id')))->result();
        if($state_list):
            $arr = array('status'=> 'true', 'message' => 'State list fetch sucessfully', 'state_list' => $state_list);
        else:
            $arr = array('status'=> 'true', 'message' => 'State list fetch Unsucessfully');
        endif;
        echo json_encode($arr);
    }
